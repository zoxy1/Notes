убрать папки или файлы от отслеживания гитом:
git rm -r --cached .idea //убирает отслеживание папки .idea 
git config --list //посмотреть настройки гита и почты 
git config --global user.name "John Doe" //установка имени используемого гитом
git config --global user.email johndoe@example.com //установка почты используемой гитом
git clone https://....git /*после этой команды с репозитория все будет скачано на локальную машину в папку DataTransfer откуда запущен гит*/
git add test.java /*если мы добавили файл в локальную папку DataTransfer, например test.txt мы должны его добавить в проект командой add*/
git add -A //добавление сразу все новые файлы в проект
/*после изменения файлов перед коммитом нужно их проиндексировать(также команда add), иначе изменения не войдут в коммит */
git status // смотрим статус нашего проекта 
git commit -m "add file test.java" // делаем коммит
git push /* закачиваем все изменения в репозиторий, вводим имя пользователя и пароль, которые были зарегины на github  */
git pull // принять все изменения с репозитория 
/*если мы выполнили 
git commit 
и больше ничего не написали, то мы попадаем в окно из которого можно выбраться 
нажав ESC затем :wq и нажать enter */
git commit --amend // изменение последнего коммита
git commit --amend -m "Новое название" /* Изменение названия последнего коммита, с той лишь разницей что нет необходимости добавлять файлы в коммит. Просто нужно указать новое название коммита
